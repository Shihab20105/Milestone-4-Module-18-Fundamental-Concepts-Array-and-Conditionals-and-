Compare Variables And Comparison Operator

Comparison

console.log(5 < 6);
/* Output:
true */
console.log(5 > 6);
/* Output:
false */
console.log(5 == 6);
/* Output:
false */
console.log(6 == 6);
/* Output:
true */
console.log(5 != 6);
/* Output:
true */
console.log(15 != 15);
/* Output:
false */

Less than or equals to

console.log(5 <= 6);
/* Output:
true */
console.log(6 <= 6);
/* Output:
true */
console.log(16 <= 6);
/* Output:
false */

Greater than or equals to

console.log(5 >= 6);
/* Output:
false */
console.log(5 >= 2);
/* Output:
true */
console.log(2 >= 2);
/* Output:
true */
